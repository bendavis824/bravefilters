! Title: Instagram Story and Post Viewer Blocker
! Description: Custom filters for Brave browser, specifically blocking Instagram story and post viewer results in Google searches
! Last modified: 2024-09-22

! Block results related to Instagram story viewers
google.*##.g:has-text(/instagram\s*(story|stories)\s*(view(er)?|watch(er)?|check(er)?|see|download(er)?)/i)
google.*##.g:has-text(/insta\s*(story|stories)\s*(view(er)?|watch(er)?|check(er)?|see|download(er)?)/i)
google.*##.g:has-text(/ig\s*(story|stories)\s*(view(er)?|watch(er)?|check(er)?|see|download(er)?)/i)

! Block results related to Instagram post viewers
google.*##.g:has-text(/instagram\s*(post|posts|photo|photos|pic|pics)\s*(view(er)?|watch(er)?|check(er)?|see|download(er)?)/i)
google.*##.g:has-text(/insta\s*(post|posts|photo|photos|pic|pics)\s*(view(er)?|watch(er)?|check(er)?|see|download(er)?)/i)
google.*##.g:has-text(/ig\s*(post|posts|photo|photos|pic|pics)\s*(view(er)?|watch(er)?|check(er)?|see|download(er)?)/i)

! Block specific domains known for Instagram viewing services
google.*##.g:has(a[href*="instagramsave"])
google.*##.g:has(a[href*="storiesig"])
google.*##.g:has(a[href*="instadp"])
google.*##.g:has(a[href*="instaviewers"])
google.*##.g:has(a[href*="storiesdown"])
