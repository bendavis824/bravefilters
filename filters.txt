! Title: Instagram Story and Post Viewer Blocker
! Description: Blocks search results for Instagram story and post viewers
! Last modified: 2024-09-22

! Block domains known for Instagram viewing services
||instagramsave.com^
||storiesig.net^
||instadp.com^
||instaviewers.com^
||storiesdown.com^

! Block search results related to Instagram story viewers
google.*##div:has-text(/instagram story viewer/i)
google.*##div:has-text(/insta story viewer/i)
google.*##div:has-text(/ig story viewer/i)

! Block search results related to Instagram post viewers
google.*##div:has-text(/instagram post viewer/i)
google.*##div:has-text(/insta post viewer/i)
google.*##div:has-text(/ig post viewer/i)

! Block ads related to Instagram viewers
##.ad-container:has-text(/instagram viewer/i)
##.sponsored-content:has-text(/story viewer/i)

! Snippet to remove Instagram viewer results (example, may need adjustment)
google.*#$#hide-if-contains-visible-text 'instagram viewer'; hide-if-contains-visible-text 'story viewer'; hide-if-contains-visible-text 'post viewer';
